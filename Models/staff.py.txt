class Staff:
    def __init__(self, sid, gid, rid, name, role, age, dob):
        self._sid = sid
        self._gid = gid
        self._rid = rid
        self.name = name
        self.role = role
        self.age = age
        self._dob = dob

    def add_reservation(self):
        print(self.name, "is adding a reservation.")

    def confirm_reservation(self):
        print(self.name, "is confirming a reservation.")

    def view_room_availability(self):
        print(self.name," is viewing room availability.")

    def modify_reservation(self):
        print(self.name," is modifying a reservation.")

    def search_reservation(self):
        print(self.name," is searching for a reservation.")

    def confirm_payment(self):
        print(self.name," is confirming a payment.")

    def check_in_guests(self):
        print(self.name," is checking in guests.")

    def check_out_guests(self):
        print(self.name," is checking out guests.")
